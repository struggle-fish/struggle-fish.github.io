类型兼容性，只要对象 objData 里包含着必须的属性即可。


interface Vegetable {
	color?: string, // ? 表示可选
	type: string
}

let getVegetables = ({ color, type }: Vegetable) => {
	return `A ${color ? color : '没值'}-${type}`
}

let objData = {
	color: '红色',
	type: '方形',
	size: 20 // 多传一个属性
}
getVegetables(objData)
